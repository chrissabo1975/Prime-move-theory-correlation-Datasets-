import pandas as pd
import numpy as np
from scipy.optimize import curve_fit
import matplotlib.pyplot as plt
import yfinance as yf

# Fetch function
def fetch_sp_data(start_date, end_date):
    data = yf.download('^GSPC', start=start_date, end=end_date)
    data = data[['Close']]
    data.rename(columns={'Close': 'close'}, inplace=True)
    return data

# Periods
periods = {
    'Black Monday 1987': ('1987-10-01', '1987-11-01'),
    '2008 Crisis': ('2008-09-01', '2008-10-31'),
    '2020 COVID Crash': ('2020-02-01', '2020-04-30'),
    '2022 Inflation Dip': ('2022-01-01', '2022-06-30')
}

# Fetch and process
datasets = {}
for name, (start, end) in periods.items():
    data = fetch_sp_data(start, end)
    data['volatility'] = data['close'].pct_change().rolling(window=5).std() * np.sqrt(252) * 100
    datasets[name] = data.dropna()

# Decay fit
def exp_decay(t, V_peak, k):
    return V_peak * np.exp(-k * t)
